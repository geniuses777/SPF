#include "pch.h"
#include <iostream>
using namespace std;

string deci_to_any(int n, int radix) {
	int r;
	string str;
	while (n != 0) {
		int i = 0;
		r = n % radix;
		n = n / radix;

		if (r > 9)
		{

			str = (char)('A' + r - 10) + str;
		}
		else
			str = (char)('0' + r) + str;
	}
	return str;
}

double power(double x, int n)
{
	double result = 1.0;
	for (int i = 0; i < n; i++)
		result = result * x;
	return result;
	// x^2
	// x^4
	// x^8
	// x^16
	// x^32
	// x^64
}

int main()
{
	cout << deci_to_any(140, 8).c_str() << endl;
	cout << deci_to_any(140, 16).c_str() << endl;
	return 0;
}
